public class Payroll 
{ 
    public static void main(String args[]) 
    { 
        double salary,stateTax, federalTax,numDependents,dependentCredit, totalWithholding, takeHomePay; 
 
        //Employee info 
        numDependents = 2; 
        salary = 1250; 
 
        //Taxes 
            stateTax = salary * 0.065; 
            federalTax = salary * 0.28; 
            totalWithholding = (stateTax + federalTax); 
        //deductions 
        dependentCredit = (numDependents * 0.025) * salary; 
 
        //final pay 
        takeHomePay = (salary - totalWithholding) + dependentCredit; 
 
        System.out.println("State Tax: $" + stateTax); 
        System.out.println("Federal Tax: $" + federalTax); 
        System.out.println("Dependents: $" + dependentCredit); 
        System.out.println("Salary: $" + salary); 
        System.out.println("Take Home Pay: $" + takeHomePay); 
        System.exit(0); 
    } 
}
